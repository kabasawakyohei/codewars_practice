////////// ///ans/// ////////// class Solution{ public static String removeDuplicateWords(String s){ StringBuilder sb = new StringBuilder(); for(String str : s.split(" ")){ if (!sb.toString().contains(str)) { sb.append(str).append(" "); } } return sb.toString().trim(); } } /// 学び/// //// split メソッドは文字列を分割して配列として新たに生成する /// split(" ")は[apple apple apple]といった配列から空白スペースごとに分割し、 /// ["apple", "apple", "apple"]という感じで３つの要素を持った配列を生成している ////////// ///test/// ////////// public class SolutionTest{ @Test public void basicTests(){ assertEquals("alpha beta gamma delta", Solution.removeDuplicateWords("alpha beta beta gamma gamma gamma delta alpha beta beta gamma gamma gamma delta")); assertEquals("my cat is fat", Solution.removeDuplicateWords("my cat is my cat fat")); } }
